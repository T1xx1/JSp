import { polyfillProposals, syntaxProposals, runtimeProposals } from './proposals';

import Table from './table.astro';

## Polyfill proposals

<Table
	pages={polyfillProposals.toSorted((a, b) => {
    	const stageA = a.frontmatter.support.tc39?.stage === 'Inactive' ? -1 : a.frontmatter.support.tc39?.stage;
    	const stageB = b.frontmatter.support.tc39?.stage === 'Inactive' ? -1 : b.frontmatter.support.tc39?.stage;

    	if (stageA !== stageB) {
    		return stageB - stageA;
    	}

    	const corejsA = a.frontmatter.support.corejs?.value === true ? 1 : 0;
    	const corejsB = b.frontmatter.support.corejs?.value === true ? 1 : 0;

    	if (corejsA !== corejsB) {
    		return corejsB - corejsA;
    	}

    	return a.frontmatter.title.localeCompare(b.frontmatter.title);
    })}
    hideCols={['proposalType']}
    variant='details'

/>
